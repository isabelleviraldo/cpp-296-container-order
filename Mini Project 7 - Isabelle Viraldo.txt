void getDate(int deliveryType){
    bool leapYear = false;
    int daysIncrease;
    if (deliveryType == 1){
        daysIncrease = 14;
    } else if (deliveryType == 2){
        daysIncrease = 5;
    } else {
        daysIncrease = 2;
    }
    
    int month = (date[1] - '0');
    month = month + (10 * (date[0] - '0'));

    int day = (date[4] - '0');
    day = day + (10 * (date[3] - '0'));
    
    int year = (date[9] - '0');
    year = year + (10 * (date[8] - '0'));
    year = year + (100 * (date[7] - '0'));
    year = year + (1000 * (date[6] - '0'));
    
    if (year % 4 == 0){
        leapYear = true;
    }
    
    int maxDays = -1;
    if (month == 4 || month == 6 || month == 9 || month == 11){
        maxDays = 30;
    } else if (month == 2){
        maxDays = 28;
        if (leapYear){
            maxDays = 29;
        }
    } else {
        maxDays = 31;
    }
    
    //new date
    
    day = day + daysIncrease;
    if (day > maxDays){
        day = day - maxDays;
        month = month + 1;
        if (month > 12){
            month = 1;
            year = year + 1;
        }
    }
    
    cout << month << "/" << day << "/" << setw(6) << year;
}
